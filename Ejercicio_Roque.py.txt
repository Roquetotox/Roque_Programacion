# Primer ejercicio
numero_1= int(input("Ingrese el primer número: "))  
numero_2 = float(input("Ingrese el segundo número:")) 
operacion = input("Ingrese la operación (suma, resta, multiplicacion, division): ")  
if operacion == "*":
    resultado = numero_1 * numero_2
    print(f"El resultado de multiplicar {numero_1} * {numero_2} es: {resultado}")

elif operacion == "suma":
    resultado = numero_1 + numero_2
    print(f"El resultado de sumar {numero_1} + {numero_2} es: {resultado}") 
elif operacion == "resta":
    resultado = numero_1 - numero_2
    print(f"El resultado de restar {numero_1} - {numero_2} es: {resultado}")    
elif operacion == "division":
    if numero_2 != 0:
        resultado = numero_1 / numero_2
        print(f"El resultado de dividir {numero_1} / {numero_2} es: {resultado}")
    else:
        print("Error: No se puede dividir por cero.")
